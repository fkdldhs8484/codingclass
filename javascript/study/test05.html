10. 다음의 결과값을 작성하시오.

function q10(arr) {
    let flag = false;
    let n = 1;
    while(!flag) {  //  while(!false)는 true...
        n++;
        for(let i = 1; i < arr.length; i++) {   // i에 들어갈 수 있는 숫자는 1~3 배열값을 대입해보면 모두 짝수 = 조건식이 true
            if((arr[0] * n) % arr[i] === 0) {   
                flag = true;
            }
            else {
                flag = false;
                break;
            }
        }
    }
    return arr[0] * n
}

console.log(q10([2,6,8,14]));   // while문이 2,6,8,14의 공통배수(최소공배수)일때 멈추므로 답은 168
