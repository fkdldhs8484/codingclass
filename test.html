08. 결괏값을 작성하시오.

    const num = [5, 9, 7, 10];

    function solution(arr) {
        let sm = arr[1], index = 1;

        for (i = 0; i < arr.length; i++) {
            if (arr[i] < sm) {
                sm = arr[i];
                index = i;
            }
        }
        arr.splice(index, 1);
        return arr;
    }
    console.log(solution(num))
}

            if (arr1[p1] == arr2[p2]) {

                p2++;
            }
            else if (arr1[p1] < arr2[p2]) p1++;
